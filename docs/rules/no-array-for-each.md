## Fail

```js

	bar(element);
});
```

```js
array.forEach((element, index) => {
	bar(element, index);
});
```

```js

	bar(element, index, array);
});
```

## Pass

```js

	bar(element);
}
```

```js
for (const [index, element] of array.entries()) {
	bar(element, index);
}
```

```js

	bar(element, index, array);
}
```
